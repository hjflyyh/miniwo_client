    <script>
      (function () {
        // 直接把 Client ID 写在这里（Cocos 不需要传参）
        const CLIENT_ID = '808005965201-9fj5v7jbfo33gkng9c4br0kb1a5gdnrs.apps.googleusercontent.com';

        // 固定回传 key（Cocos 只需监听/读取这个 key）
        const STORAGE_KEY = 'miniwo_google_login:last';

        const $status = document.getElementById('status');
        const setStatus = (msg) => { if ($status) $status.textContent = msg; };
        document.getElementById('closeBtn').addEventListener('click', () => { try { window.close(); } catch(e) {} });

        function notifyToGame(payload) {
          // 1) Web: 尝试 postMessage（iframe/跨域也可用）
          try {
            if (window.parent && window.parent !== window) {
              window.parent.postMessage({ type: 'GOOGLE_LOGIN', payload }, '*');
            }
          } catch (e) {}

          // 2) Native: 通过 URL scheme 触发 setOnJSCallback（可选）
          try {
            const q = new URLSearchParams();
            q.set('id_token', payload.id_token || '');
            if (payload.select_by) q.set('select_by', payload.select_by);
            document.location = 'miniwo://' + q.toString();
          } catch (e) {}

          // 3) 兜底：localStorage（Web 上也方便轮询）
          try {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(payload));
          } catch (e) {}
        }

        function onCredentialResponse(resp) {
          try {
            if (!resp || !resp.credential) {
              setStatus('未获取到 credential（id_token）。');
              return;
            }
            const payload = {
              id_token: resp.credential,
              select_by: resp.select_by,
              ts: Date.now()
            };
            setStatus('登录成功，正在回传…');
            notifyToGame(payload);
          } catch (e) {
            setStatus('登录回调处理失败：' + (e && e.message ? e.message : String(e)));
          }
        }

        function initWhenReady() {
          try {
            if (!window.google || !google.accounts || !google.accounts.id) {
              setTimeout(initWhenReady, 50);
              return;
            }
            google.accounts.id.initialize({
              client_id: CLIENT_ID,
              callback: onCredentialResponse,
              auto_select: false,
              cancel_on_tap_outside: false,
            });
            google.accounts.id.renderButton(
              document.getElementById('button'),
              { theme: 'outline', size: 'large', shape: 'rectangular', text: 'signin_with', width: 280 }
            );
            setStatus('请点击按钮登录。');
          } catch (e) {
            setStatus('初始化失败：' + (e && e.message ? e.message : String(e)));
          }
        }

        if (!CLIENT_ID) {
          setStatus('CLIENT_ID 为空，请在 google-login.txt 中填写。');
          return;
        }
        initWhenReady();
      })();
    </script>
